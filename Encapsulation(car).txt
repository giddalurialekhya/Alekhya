public class ExampleofEncapsulation {
    public static void main(String[] args) {
        Car c = new Car();
        c.setData("Thar", "Black", 200);

        System.out.println("The name of car: " + c.getName());
        System.out.println("The color of car: " + c.getColor());
        System.out.println("The speed of car: " + c.getSpeed());
    }
}

class Car {
    private String name;
    private String color;
    private int speed;

    public void setData(String x, String y, int z) {
        name = x;
        color = y;
        speed = z;
    }

    public String getName() {
        return name;
    }

    public String getColor() {
        return color;
    }

    public int getSpeed() {
        return speed;
    }
}
